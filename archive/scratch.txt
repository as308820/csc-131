import { useState, useEffect } from 'react';

function App() {
  const [data, setData] = useState("Hello World");
  const [isLoggedIn, setIsLoggedIn] = useState(false);

  useEffect(() => {
    getTest()
      .then((res) => {
        setData(res.message);
      })
      .catch((err) => console.log(err));    
  }, []);

  return (
    <div className="App">
      {!isLoggedIn ? (
        <LoginForm />
      ) : (

      )}
    </div>
  );
}

export default App;
